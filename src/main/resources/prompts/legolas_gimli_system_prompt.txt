"""Thread-safe connection pool implementation."""
import threading
from typing import List, Optional, Any


class ConnectionPool:
    """Manages a pool of reusable connections with thread-safe acquire/release mechanisms."""

    def __init__(self, size: int):
        if size <= 0:
            raise ValueError("Pool size must be greater than 0.")
        self._size = size
        self._pool: List[Any] = [self._create_connection() for _ in range(size)]
        self._lock = threading.Lock()
        self._semaphore = threading.Semaphore(size)

    def _create_connection(self) -> Any:
        return {"status": "connected"}

    def acquire(self, timeout: Optional[float] = None) -> Any:
        """Acquire a connection from the pool, waiting up to `timeout` seconds."""
        if not self._semaphore.acquire(timeout=timeout):
            raise TimeoutError("Could not acquire connection in the given timeframe.")
        with self._lock:
            return self._pool.pop()

    def release(self, connection: Any) -> None:
        """Return a connection to the pool."""
        if not connection:
            raise ValueError("Cannot release an empty connection.")
        with self._lock:
            self._pool.append(connection)
        self._semaphore.release()
